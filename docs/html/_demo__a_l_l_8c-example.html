// we assume a 16 char, 2 line display (1602 model)
#define LCD_ADDRESS 0x27
#define LCD_COLUMNS 16
#define LCD_ROWS 2
void slowWrite(const char str[], uint32_t msDelay ) {
char c;
if (str != NULL) while( c = *str++) {
lcd_writeChar(c);
if(msDelay != 0) sleep_ms(msDelay);
}
}
